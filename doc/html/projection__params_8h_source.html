33 class SpanParams {
34  public:
38  enum class Direction { HORIZONTAL, VERTICAL };
39 
40  SpanParams() {}
41  SpanParams(const Radians& start_angle, const Radians& end_angle,
42  const Radians& step) {
43  _start_angle = start_angle;
44  _end_angle = end_angle;
45  _step = step;
46  _num_beams = floor((_end_angle - _start_angle) / _step);
47  _span = Radians::Abs(end_angle - start_angle);
48  }
49 
50  SpanParams(const Radians& start_angle, const Radians& end_angle,
51  int num_beams) {
52  _start_angle = start_angle;
53  _end_angle = end_angle;
54  _num_beams = num_beams;
55  _step = (_end_angle - _start_angle) / _num_beams;
56  _span = Radians::Abs(end_angle - start_angle);
57  }
58 
59  const Radians& start_angle() const { return _start_angle; }
60  const Radians& end_angle() const { return _end_angle; }
61  const Radians& step() const { return _step; }
62  const Radians& span() const { return _span; }
63  int num_beams() const { return _num_beams; }
64 
65  bool valid() const { return _num_beams > 0 && _span > 0_deg; }
66 
67  private:
68  Radians _start_angle = 0_deg;
69  Radians _end_angle = 0_deg;
70  Radians _step = 0_deg;
71  Radians _span = 0_deg;
72  int _num_beams = 0;
73 };

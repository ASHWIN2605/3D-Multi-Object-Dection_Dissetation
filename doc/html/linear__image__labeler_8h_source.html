40 template <int16_t STEP_ROW = 1, int16_t STEP_COL = 1>
41 class LinearImageLabeler : public AbstractImageLabeler {
42  public:
43  static constexpr int16_t NEIGH_SIZE = 2 * STEP_ROW + 2 * STEP_COL;
44  std::array<PixelCoord, NEIGH_SIZE> Neighborhood;

53  explicit LinearImageLabeler(const cv::Mat& depth_image,
54  const ProjectionParams& params,
55  const Radians& angle_threshold)
56  : AbstractImageLabeler(depth_image, params, angle_threshold) {
57  // this can probably be done at compile time
58  int16_t counter = 0;
59  for (int16_t r = STEP_ROW; r > 0; --r) {
60  Neighborhood[counter++] = PixelCoord(-r, 0);
61  Neighborhood[counter++] = PixelCoord(r, 0);
62  }
63  for (int16_t c = STEP_COL; c > 0; --c) {
64  Neighborhood[counter++] = PixelCoord(0, -c);
65  Neighborhood[counter++] = PixelCoord(0, c);
66  }
67  }
68  virtual ~LinearImageLabeler() {}

78  void LabelOneComponent(uint16_t label, const PixelCoord& start,
79  const AbstractDiff* diff_helper) {
80  // breadth first search
81  std::queue<PixelCoord> labeling_queue;
82  labeling_queue.push(start);
83  // while the queue is not empty continue removing front point adding its
84  // neighbors back to the queue - breadth-first-search one component
85  size_t max_queue_size = 0;
86  while (!labeling_queue.empty()) {
87  max_queue_size = std::max(labeling_queue.size(), max_queue_size);
88  // copy the current coordinate
89  const PixelCoord current = labeling_queue.front();
90  labeling_queue.pop();
91  uint16_t current_label = LabelAt(current);
92  if (current_label > 0) {
93  // we have already labeled this point. No need to add it.
94  continue;
95  }
96  // set the label of this point to current label
97  SetLabel(current, label);
98 
99  // check the depth
100  auto current_depth = DepthAt(current);
101  if (current_depth < 0.001f) {
102  // depth of this point is wrong, so don't bother adding it to queue
103  continue;
104  }
105  for (const auto& step : Neighborhood) {
106  PixelCoord neighbor = current + step;
107  if (neighbor.row < 0 || neighbor.row >= _label_image.rows) {
108  // point doesn't fit
109  continue;
110  }
111  // if we just went over the borders in horiz direction - wrap around
112  neighbor.col = WrapCols(neighbor.col);
113  uint16_t neigh_label = LabelAt(neighbor);
114  if (neigh_label > 0) {
115  // we have already labeled this one
116  continue;
117  }
118  auto diff = diff_helper->DiffAt(current, neighbor);
119  if (diff_helper->SatisfiesThreshold(diff, _radians_threshold)) {
120  labeling_queue.push(neighbor);
121  }
122  }
123  }
124  }

160  int16_t WrapCols(int16_t col) const {
161  // we allow our space to fold around cols
162  if (col < 0) {
163  return col + _label_image.cols;
164  }
165  if (col >= _label_image.cols) {
166  return col - _label_image.cols;
167  }
168  return col;
169  }

174  void ComputeLabels(DiffFactory::DiffType diff_type) override {
175  _label_image =
176  cv::Mat::zeros(_depth_image_ptr->size(), cv::DataType<uint16_t>::type);
177  auto diff_helper_ptr =
178  DiffFactory::Build(diff_type, _depth_image_ptr, &_params);
179  // initialize the label
180  uint16_t label = 1;
181 
182  for (int row = 0; row < _label_image.rows; ++row) {
183  for (int col = 0; col < _label_image.cols; ++col) {
184  if (_label_image.at<uint16_t>(row, col) > 0) {
185  // we have already labeled this point
186  continue;
187  }
188  if (_depth_image_ptr->at<float>(row, col) < 0.001f) {
189  // depth is zero, not interested
190  continue;
191  }
192  LabelOneComponent(label, PixelCoord(row, col), diff_helper_ptr.get());
193  // we have finished labeling this connected component. We now need to
194  // label the next one, so we increment the label
195  label++;
196  }
197  }
198  }

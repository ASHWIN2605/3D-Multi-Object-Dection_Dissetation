65  void OnNewObjectReceived(const Cloud& cloud, const int sender_id) override {
66  auto pcl_cloud_ptr = cloud.ToPcl();
67  std::unordered_map<uint16_t, Cloud> clusters;
68  if (this->_counter++ % this->_skip != 0) {
69  // share empty clusters
70  this->ShareDataWithAllClients(clusters);
71  return;
72  }
73  auto start = high_resolution_clock::now();
74  typename pcl::search::KdTree<PointT>::Ptr tree(
75  new pcl::search::KdTree<PointT>);
76  tree->setInputCloud(pcl_cloud_ptr);
77  std::vector<pcl::PointIndices> cluster_indices;
78  pcl::EuclideanClusterExtraction<PointT> clusterer;
79  clusterer.setClusterTolerance(this->_cluster_tollerance);
80  clusterer.setMinClusterSize(this->_min_cluster_size);
81  clusterer.setMaxClusterSize(this->_max_cluster_size);
82  clusterer.setSearchMethod(tree);
83  clusterer.setInputCloud(pcl_cloud_ptr);
84  clusterer.extract(cluster_indices);
85 
86  auto end = high_resolution_clock::now();
87  fprintf(stderr, "euclidian based labeling took: %lu us\n",
88  std::chrono::duration_cast<microseconds>(end - start).count());
89  for (auto cluster_iter = cluster_indices.begin();
90  cluster_iter != cluster_indices.end(); ++cluster_iter) {
91  int idx = std::distance(cluster_indices.begin(), cluster_iter);
92  Cloud cloud_cluster(cloud.pose());
93  cloud_cluster.reserve(cluster_iter->indices.size());
94  for (auto point_iter = cluster_iter->indices.begin();
95  point_iter != cluster_iter->indices.end(); ++point_iter) {
96  cloud_cluster.push_back(cloud[*point_iter]);
97  }
98  clusters[idx] = cloud_cluster;
99  }
100  this->ShareDataWithAllClients(clusters);
101  }
102 };

96  void OnNewObjectReceived(const Cloud& cloud, int) override {
97  // generate a projection from a point cloud
98  if (!cloud.projection_ptr()) {
99  fprintf(stderr, "ERROR: projection not initialized in cloud.\n");
100  fprintf(stderr, "INFO: cannot label this cloud.\n");
101  return;
102  }
103  time_utils::Timer timer;
104  LabelerT image_labeler(cloud.projection_ptr()->depth_image(),
105  cloud.projection_ptr()->params(), _angle_tollerance);
106  image_labeler.ComputeLabels(_diff_type);
107  const cv::Mat* labels_ptr = image_labeler.GetLabelImage();
108  fprintf(stderr, "INFO: image based labeling took: %lu us\n",
109  timer.measure());
110 
111  // send image to whoever wants to get it
112  if (_label_client) {
113  _label_client->OnNewObjectReceived(*labels_ptr, this->id());
114  }
115 
116  fprintf(stderr, "INFO: labels image sent to clients in: %lu us\n",
117  timer.measure());
118 
119  // create 3d clusters from image labels
120  std::unordered_map<uint16_t, Cloud> clusters;
121  for (int row = 0; row < labels_ptr->rows; ++row) {
122  for (int col = 0; col < labels_ptr->cols; ++col) {
123  const auto& point_container = cloud.projection_ptr()->at(row, col);
124  if (point_container.IsEmpty()) {
125  // this is ok, just continue, nothing interesting here, no points.
126  continue;
127  }
128  uint16_t label = labels_ptr->at<uint16_t>(row, col);
129  if (label < 1) {
130  // this is a default label, skip
131  continue;
132  }
133  for (const auto& point_idx : point_container.points()) {
134  const auto& point = cloud.points()[point_idx];
135  clusters[label].push_back(point);
136  }
137  }
138  }
139 
140  // filter out unfitting clusters
141  std::vector<uint16_t> labels_to_erase;
142  for (const auto& kv : clusters) {
143  const auto& cluster = kv.second;
144  if (cluster.size() < this->_min_cluster_size ||
145  cluster.size() > this->_max_cluster_size) {
146  labels_to_erase.push_back(kv.first);
147  }
148  }
149  for (auto label : labels_to_erase) {
150  clusters.erase(label);
151  }
152 
153  fprintf(stderr, "INFO: prepared clusters in: %lu us\n", timer.measure());
154 
155  this->ShareDataWithAllClients(clusters);
156  fprintf(stderr, "INFO: clusters shared: %lu us\n", timer.measure());
157  }
158 
159  private:
160  int _counter;
161  Radians _angle_tollerance;
162 
163  AbstractClient<cv::Mat>* _label_client;
164 
165  DiffFactory::DiffType _diff_type = DiffFactory::DiffType::NONE;
166 };

39 template <int16_t STEP_ROW = 1, int16_t STEP_COL = 1>
40 class DijkstraImageLabeler : public AbstractImageLabeler {
41  public:
42  struct PixelCoord {
43  PixelCoord() : row(0), col(0) {}
44  PixelCoord(int16_t row_, int16_t col_) : row(row_), col(col_) {}
45 
46  int16_t row;
47  int16_t col;
48  };
49 
50  struct WeightedPixelCoord : public PixelCoord {
51  WeightedPixelCoord() : PixelCoord(), keep_label(true) {}
52  WeightedPixelCoord(int16_t row_, int16_t col_, bool keep_label_)
53  : PixelCoord(row_, col_), keep_label(keep_label_) {}
54 
55  bool operator<(const WeightedPixelCoord& other) const {
56  return keep_label < other.keep_label;
57  }
58 
59  bool keep_label;
60  };
61 
62  static constexpr int16_t NEIGH_SIZE = 2 * STEP_ROW + 2 * STEP_COL;
63  std::array<PixelCoord, NEIGH_SIZE> Neighborhood;
64 
65  explicit DijkstraImageLabeler(const cv::Mat& depth_image,
66  const ProjectionParams& params,
67  const Radians& angle_threshold)
68  : AbstractImageLabeler(depth_image, params, angle_threshold) {
69  // this can probably be done in compile time
70  int16_t counter = 0;
71  for (int16_t r = STEP_ROW; r > 0; --r) {
72  Neighborhood[counter++] = PixelCoord(-r, 0);
73  Neighborhood[counter++] = PixelCoord(r, 0);
74  }
75  for (int16_t c = STEP_COL; c > 0; --c) {
76  Neighborhood[counter++] = PixelCoord(0, -c);
77  Neighborhood[counter++] = PixelCoord(0, c);
78  }
79  }
80 
81  virtual ~DijkstraImageLabeler() {}
82 
83  void ComputeLabels(DiffFactory::DiffType diff_type) override {
84  _label_image =
85  cv::Mat::zeros(_depth_image_ptr->size(), cv::DataType<uint16_t>::type);
86  // initialize the label
87  uint16_t label = 1;
88 
89  std::priority_queue<WeightedPixelCoord> labeling_queue;
90  labeling_queue.push(WeightedPixelCoord());
91  // while the queue is not empty continue removing front point adding its
92  // neighbors back to the queue - breadth-first-search one component
93  while (!labeling_queue.empty()) {
94  // copy the current coordinate
95  const WeightedPixelCoord current = labeling_queue.top();
96  labeling_queue.pop();
97 
98  uint16_t& current_label =
99  _label_image.at<uint16_t>(current.row, current.col);
100  if (current_label > 0) {
101  // we have already labeled this point. No need to add it.
102  continue;
103  }
104 
105  // this point hints it is time to change the label
106  if (!current.keep_label) {
107  label++;
108  }
109 
110  // set the label of this point to current label
111  current_label = label;
112  auto current_depth =
113  _depth_image_ptr->at<float>(current.row, current.col);
114  if (current_depth < 0.001f) {
115  // we have already labeled this point. No need to add it.
116  continue;
117  }
118  for (const auto& step : Neighborhood) {
119  int16_t neighbor_row = current.row + step.row;
120  if (neighbor_row < 0 || neighbor_row >= _label_image.rows) {
121  // point doesn't fit
122  continue;
123  }
124  int16_t neighbor_col = current.col + step.col;
125  if (neighbor_col < 0) {
126  neighbor_col += _label_image.cols;
127  }
128  if (neighbor_col >= _label_image.cols) {
129  neighbor_col -= _label_image.cols;
130  }
131  if (_label_image.at<uint16_t>(neighbor_row, neighbor_col) > 0) {
132  continue;
133  }
134  float alpha;
135  if (step.col == 0) {
136  // means that there is a row step
137  alpha = (_params.AngleFromRow(neighbor_row) -
138  _params.AngleFromRow(current.row)).val();
139  } else {
140  alpha = (_params.AngleFromCol(neighbor_col) -
141  _params.AngleFromCol(current.col)).val();
142  }
143  const float& neighbor_depth =
144  _depth_image_ptr->at<float>(neighbor_row, neighbor_col);
145  labeling_queue.push(WeightedPixelCoord(
146  neighbor_row, neighbor_col,
147  BelongToOneCluster(current_depth, neighbor_depth, alpha)));
148  }
149  }
150  }
151 };
